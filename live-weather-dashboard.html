Python-Tutorial: Weather-Dashboard

This is a simple weather dashboard written in Python.

import requests
import pytz
from rich.console import Console
from rich.table import Table

console = Console()
cities = ["Delhi", "London", "Berlin", "New York", "Tokyo", "Oslo", "Moscow"]
table = Table(title="Weather Dashboard", style="cyan")
table.add_column("City", justify="center")
table.add_column("Temperature (Â°C)", justify="center")
for city in cities:
    temp = requests.get(f"https://wttr.in/{city}?format=%t").text
    table.add_row(city, temp)
    
console.print(table)
